        }
        break;

        this.setState({gender: value}, () => {console.log(this.state)});
        break;
      case "day":
        birthDate.day=value;
        break;
      case "month":
        birthDate.month=value;
        break;
      case "year":
        birthDate.year=value;
        break;                
    }
